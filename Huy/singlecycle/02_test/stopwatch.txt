# delay1s:
# ---------------------------------
delay1s:
    li x3, 0
    li x4, 0x3F940B   # ~1s @25MHz
delay_loop:
    jal x18, detectkey
    addi x3, x3, 1
    bne  x3, x4, delay_loop
    jalr x0, x1, 0

# ---------------------------------
# add1to9tohexled:
# ---------------------------------
add1tohex:
    addi x7, x7, 1
    sw x7, 0(x5)
    jal x1, delay1s
    blt x7, x8, add1tohex
    jalr x0, x2, 0
    
# ---------------------------------
# detectkey:
# ---------------------------------
detectkey:
    lw x9, 0(x6)
    bne x9, x10, detectkey
    jalr x0, x18, 0
